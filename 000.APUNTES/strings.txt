#include <stdio.h>
#include <string.h>

int main() {
   //declarar e inicializar (en estos casos no hace falta el '/0')
   char cadena[] = {'h', 'o', 'l', 'a', '\n'};
   char cadena[] = "hola\n";
   char* cadena = "hola\n";

   //declarar, luego inicializar (pedimos memoria estática)
   char cadena[5];
   cadena[0] = 'h';
   cadena[1] = 'o';
   cadena[2] = 'l';
   cadena[3] = 'a';
   cadena[4] = '\n';
   cadena[5] = '\0'; //importante para terminar de leer el string

   //declarar, luego inicializar (pedimos memoria dinámica)
   char* cadena = malloc(sizeof(char) * 7);
   cadena[0] = 'h';
   cadena[1] = 'o';
   cadena[2] = 'l';
   cadena[3] = 'a';
   cadena[4] = '\n';
   cadena[5] = '\0';

   printf("%s", cadena);
   printf("%zu \n", strlen(cadena)); //size_t

   free(cadena) //cuando usemos memoria dinámica

   return 0;
}



NOTAS:

-  char* cadena; cadena = "nombre123";   funciona pero al pisar el valor se pierde la referencia al string original (memory leak). No se recomienda usar esta forma. Mejor usar strcpy() o equivalentes para pasar cadenas enteras.
