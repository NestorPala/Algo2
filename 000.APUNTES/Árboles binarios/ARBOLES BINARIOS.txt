* Para todos los arboles el caso base de la altura es:

struct ab;

size_t max(a,b);

size_t ab_altura(const ab_t* arbol) {
	if (arbol == NULL) {  //caso base (NO OLVIDAR)
		return 0;
	}

	return max(ab_altura(arbol->izq), ab_altura(arbol->der)) + 1;

	return 0;
}
